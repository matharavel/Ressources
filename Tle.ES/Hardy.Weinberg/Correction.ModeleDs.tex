\noindent \textbf{Exercice 1.}
\vskip 0.2 cm
\noindent
\begin{enumerate}
\item  
\begin{enumerate}
\item %Déterminer la proportion $p$ de personnes favorables à cette construction parmi les personnes interrogées.
On a~$p=\dfrac{700}{\nombre{1100}}\simeq 0,6364$

\item %Déterminer, au niveau de confiance de 95\,\%  un intervalle de confiance de la proportion de personnes favorables à cette construction.

$I=\left[p-\dfrac{1}{\sqrt{n}}~;~p+\dfrac{1}{\sqrt{n}}\right]\simeq [0,60~;~0,67]$

\item %La municipalité trouve que la marge d'erreur (amplitude) de cet intervalle est trop grande.\\
%Il souhaiterait que cette marge soit inférieure ou égale à 0.04.\\
%Combien de personnes faut-il interroger ?
La marge d'erreur est $\dfrac{2}{\sqrt{n}}$.\\
On cherche $n$ tel que $\dfrac{2}{\sqrt{n}}\leqslant 0,04$. On en déduit $\dfrac{1}{\sqrt{n}}\leqslant 0,02$ d'où $\sqrt{n}\geqslant \dfrac{1}{0,02}$ \\
puis $n\geqslant \left(\dfrac{1}{0,02}\right)^2=50^{2}=\nombre{2500}$.\\
Il faut interroger au moins \nombre{2500} personnes.
\end{enumerate}


\end{enumerate}

\noindent \textbf{Exercice 2.}
\vskip 0.2 cm
\noindent

\begin{enumerate}
\item %Calculer la proportion d’animaux marqués lors de la recapture.
La proportion d'animaux marqués lors de la recapture est $p=\dfrac{375}{1500}=0,25$.

\item %Estimer l’abondance de l’espèce étudiée.
Pour estimer l'abondance de l'espèce étudiée, on utilise la méthode CMR.\\
En notant $N$ l'effectif cherché, $M=1200$, $n=1500$, $m=375$, on a : \\
$N=M\times \dfrac{n}{m}=1200\times \dfrac{1500}{375}=\boxed{\textcolor{red}{\nombre{4800}}}$

\item 
\begin{enumerate}
\item %Déterminer un intervalle de confiance au seuil de confiance 95\:\% de la proportiond’animaux marqués.\\
Les conditions sont réunies pour utiliser un intervalle de confiance au seuil de confiance 95\:\%.\\
$I=\left[p-\dfrac{1}{\sqrt{n}}~;~p+\dfrac{1}{\sqrt{n}}\right]:\left[0,25-\dfrac{1}{\sqrt{\numprint{1500}}}~;~0,25+\dfrac{1}{\sqrt{\numprint{1500}}}\right]\simeq \boxed{\textcolor{red}{[0,224~;~0,276]}}$.

\item %En déduire un encadrement de l’abondance de l’espèce étudiée.
$0,224\leqslant p\leqslant 0,276\Rightarrow \dfrac{1}{0,276}\leqslant \dfrac{1}{p}\leqslant \dfrac{1}{0,224}\Rightarrow \dfrac{1200}{0,276}\leqslant N\leqslant \dfrac{1200}{0,224}$ donc $\boxed{\textcolor{red}{\numprint{4347}\leqslant N\leqslant \nombre{5358}}}$.
\end{enumerate}
\end{enumerate}

\bigskip
